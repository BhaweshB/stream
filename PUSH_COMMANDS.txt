# Copy and paste these commands in Git Bash or Command Prompt

# 1. Check status
git status

# 2. Add remote (if not already added)
git remote add origin git@github.com:BhaweshB/stream.git

# 3. Add all files
git add .

# 4. Commit
git commit -m "Initial commit: Real-time streaming platform with low-latency support"

# 5. Push to GitHub
git push -u origin main

# If main doesn't work, try master:
# git push -u origin master

# 6. Verify on GitHub
# Open: https://github.com/BhaweshB/stream
